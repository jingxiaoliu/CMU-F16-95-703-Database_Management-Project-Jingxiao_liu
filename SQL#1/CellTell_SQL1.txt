CREATE TABLE customer 
(ID          NUMBER(5), 
 LName       VARCHAR2(30), 
 FName       VARCHAR2(30), 
 street      VARCHAR2(30), 
 city        VARCHAR2(30), 
 zipcode     NUMBER(5),
 state       VARCHAR2(2), 
 phone       VARCHAR2(12), 
 CreditScore NUMBER(3), 
 credit_org  VARCHAR2(30), 
 CS_date     DATE, 
 CONSTRAINT customer_ID_PK PRIMARY KEY (ID)  ); 
  
CREATE TABLE PLAN 
(plan_ID     NUMBER(6), 
 Start_Date  DATE DEFAULT SYSDATE, 
 End_Date    DATE, 
 BasePrice   NUMBER(3), 
 plan_type   VARCHAR2(5), 
 CONSTRAINT plan_planID_PK PRIMARY KEY (plan_ID), 
 CONSTRAINT plan_dates_CK    CHECK (Start_Date <= End_Date), 
 CONSTRAINT plan_plantype_CK CHECK (plan_type IN ('voice', 'data', 'text') AND Plan_type IS NOT NULL)   ); 

CREATE TABLE VOICE 
(vplan_ID     NUMBER(6), 
 nummin       NUMBER(5), 
 hasvoicemail VARCHAR2(3), 
 hasthreeway  VARCHAR2(3), 
 overagefee   NUMBER(4,2) DEFAULT 0, 
 CONSTRAINT voice_vplanID_PK PRIMARY KEY (vplan_ID), 
 CONSTRAINT voice_vplanID_FK FOREIGN KEY (vplan_ID) REFERENCES plan (plan_ID), 
 CONSTRAINT voice_hasvoicemail_CK CHECK (hasvoicemail IN ('Yes', 'No') AND hasvoicemail IS NOT NULL), 
 CONSTRAINT voice_hasthreeway_CK  CHECK (hasthreeway  IN ('Yes', 'No') AND hasthreeway  IS NOT NULL)   ); 
  
CREATE TABLE TEXT 
(tplan_ID    NUMBER(6), 
 textlimit   NUMBER(5), 
 hasmms      VARCHAR2(3), 
 mmsoverage  NUMBER(4,2) DEFAULT 0, 
 textoverage NUMBER(4,2) DEFAULT 0, 
 CONSTRAINT text_tplanID_PK PRIMARY KEY (tplan_ID), 
 CONSTRAINT text_tplanID_FK FOREIGN KEY (tplan_ID) REFERENCES plan (plan_ID), 
 CONSTRAINT text_hasmms_CK  CHECK (hasmms IN ('Yes', 'No') AND hasmms IS NOT NULL)   ); 

CREATE TABLE DATA 
(dPlan_ID             NUMBER(6), 
 dataLimit            NUMBER(5), 
 NetworkTechnology    VARCHAR2(2), 
 overagefee NUMBER(4,2) DEFAULT 0, 
 CONSTRAINT data_dplanID_PK PRIMARY KEY (dplan_ID), 
 CONSTRAINT data_dplanID_FK FOREIGN KEY (dplan_ID) REFERENCES plan (plan_ID),
 CONSTRAINT data_network_CK CHECK (NetworkTechnology IN ('No','2G', '3G', '4G') AND NetworkTechnology IS NOT NULL)   );
  
CREATE TABLE CONTRACT 
(contract_ID      NUMBER(5), 
 customer_ID      NUMBER(5) NOT NULL, 
 vPlan_ID         NUMBER(6) NOT NULL, 
 tPlan_ID         NUMBER(6), 
 dPlan_ID         NUMBER(6), 
 Start_Date       DATE DEFAULT SYSDATE, 
 End_Date         DATE, 
 discountperc     NUMBER(2), 
 CONSTRAINT contract_ID_PK PRIMARY KEY (contract_ID), 
 CONSTRAINT contract_customerID_FK FOREIGN KEY (customer_ID) REFERENCES customer (ID), 
 CONSTRAINT contract_voice_FK      FOREIGN KEY (vplan_ID)    REFERENCES voice (vplan_ID), 
 CONSTRAINT contract_text_FK       FOREIGN KEY (tplan_ID)    REFERENCES text  (tplan_ID), 
 CONSTRAINT contract_data_FK       FOREIGN KEY (dplan_ID)    REFERENCES data  (dplan_ID)  ); 

CREATE TABLE MANUFACTURER 
(manuf_ID       NUMBER(6), 
 manuf_name     VARCHAR2(30), 
 contact_lname  VARCHAR2(30), 
 contact_fname  VARCHAR2(30), 
 contact_phone  NUMBER(15), 
 contact_email  VARCHAR2(50), 
 CONSTRAINT manufacturer_manufID_PK PRIMARY KEY (manuf_ID), 
 CONSTRAINT manufacturer_contactemail_CK CHECK (contact_email LIKE '%@%.%') ); 
  
CREATE TABLE PRODUCT 
(product_ID NUMBER(5), 
 name       VARCHAR2(30), 
 manuf_ID   NUMBER(6), 
 costpaid   NUMBER(6,2) NOT NULL, 
 baseprice  NUMBER(6,2) NOT NULL, 
 Type       VARCHAR2(9), 
 CONSTRAINT product_ID_PK PRIMARY KEY (product_ID), 
 CONSTRAINT product_manufID_FK FOREIGN KEY (manuf_ID) REFERENCES manufacturer (manuf_ID), 
 CONSTRAINT product_type_CK    CHECK (Type IN ('CellPhone', 'Accessory') AND Type IS NOT NULL) ); 
 
CREATE TABLE ACCESSORY 
(access_ID  NUMBER(5), 
 category   VARCHAR2(30), 
 CONSTRAINT accessory_accessID_PK PRIMARY KEY (access_ID), 
 CONSTRAINT accessory_accessID_FK FOREIGN KEY (access_ID) REFERENCES product (product_ID)   ); 
  
CREATE TABLE CELLPHONE 
(phone_ID           NUMBER(5), 
 Model              VARCHAR2(30), 
 NetworkTechnology  VARCHAR2(4), 
 OS                 VARCHAR2(20),
 CONSTRAINT cellphone_phoneID_PK PRIMARY KEY (phone_ID), 
 CONSTRAINT cellphone_phoneID_FK FOREIGN KEY (phone_ID) REFERENCES product (product_ID)  ); 
  
CREATE TABLE PHONE_ACCESS 
(phone_ID   NUMBER(5), 
 access_ID  NUMBER(5), 
 CONSTRAINT phone_access_PK PRIMARY KEY (phone_ID, access_ID), 
 CONSTRAINT phone_access_phoneID_FK  FOREIGN KEY (phone_ID)  REFERENCES cellphone (phone_ID), 
 CONSTRAINT phone_access_accessID_FK FOREIGN KEY (access_ID) REFERENCES accessory (access_ID) ); 
  
CREATE TABLE CONTRACT_CELLPHONE 
(contract_ID NUMBER(5), 
 phone_ID    NUMBER(5), 
 seq#        NUMBER(1), 
 paidprice   NUMBER(6,2) NOT NULL, 
 ESN         NUMBER(10)  NOT NULL, 
 CONSTRAINT contract_cellphone_PK PRIMARY KEY (contract_ID, phone_ID, seq#), 
 CONSTRAINT contract_cellphone_phoneID_FK FOREIGN KEY (phone_ID)    REFERENCES cellphone (phone_ID), 
 CONSTRAINT contract_cellphone_contrID_FK FOREIGN KEY (contract_ID) REFERENCES contract (contract_ID) );
