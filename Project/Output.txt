SQL> SELECT L || R AS semester,
  2         LAG(grade, 1) OVER (ORDER BY R, L DESC) AS "Previous Semester",
  3     grade AS "Current Semester",
  4         LEAD(grade, 1) OVER (ORDER BY R, L DESC) AS "Following Semester"
  5  FROM(
  6  SELECT SUBSTR(semester,1,INSTR(semester,' ')) AS L, SUBSTR(semester,INSTR(semester,' ')+1,INSTR(semester,' ')+4) AS R, grade
  7  FROM(
  8  SELECT DISTINCT sem_Cmpleted AS semester, ROUND(AVG(score),2) AS grade
  9  FROM   training JOIN grade_score USING(grade)
 10  GROUP BY sem_Cmpleted))
 11  ORDER BY R, L DESC;
